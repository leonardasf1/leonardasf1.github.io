package main
import "fmt"

func main() {
  arr := [7]int{5, 8, 3, 6, 0, 1, 2}
  arr1 := arr
  count := 0
  for i := 0; i <= len(arr); i++ {
    for j := 1; j < len(arr); j++ {
      if arr[j-1] > arr[j] {
        arr[j-1], arr[j] = arr[j], arr[j-1]
      }
      count++
    }
    if arr1 == arr {
      fmt.Println("done", arr, count)
      return
    }
    arr1 = arr
  }
}

package main

import "testing"

var globalVarN = 5
var globalVarResult = 120

func TestFactorial(t *testing.T) {
  result, i := 1, 1

  for ; i <= globalVarN; i++ {
    result *= i
    println(result)
  }

  if result != globalVarResult {
    t.Errorf("Factorial expected %v but got %v", globalVarResult, result)
  }
}
