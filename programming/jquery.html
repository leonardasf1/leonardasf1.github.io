$(window).on("scroll", function () {
    var scr = $(this).scrollTop();
    if( scr > 107 ) {
        $('.content').addClass('scrolled');
    }   
    if( scr <= 107 ) {     
        $('.content').removeClass('scrolled');
    }
});

$(window).scroll(function () {
    var scrolled = $(this).scrollTop();
    if( scrolled >= 200 ) {
        $('.content').addClass('scrolled');
    }   
    else {     
        $('.content').removeClass('scrolled');
    }
});
